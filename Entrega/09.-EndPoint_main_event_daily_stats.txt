from datetime import date
from fastapi import FastAPI, HTTPException
import json
from typing import Optional, Any
app = FastAPI()

def event_daily_stats() -> Any:

    file_path = r'.venv\08.-EndPoint_event_daily_stats.json'

    try:
        with open(file_path, "r", encoding="utf-8") as f:
            data = json.load(f)
        return data
    except FileNotFoundError:
        raise HTTPException(status_code=404, detail="No existe archivo")
    except json.JSONDecodeError:
        raise HTTPException(status_code=500, detail="Error al decodificar el archivo.JSON")
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
