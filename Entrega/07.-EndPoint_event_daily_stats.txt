%sql
INSERT OVERWRITE DIRECTORY '/Volumes/testdatabricks/default/datatest/EndPointevent_daily_stats'
USING json
OPTIONS (maptypesjson 'true')

WITH users AS
(
    SELECT DISTINCT user_id,date_format(timestamp, 'yyyy-MM-dd') AS Date FROM purchase
    UNION
    SELECT DISTINCT user_id,date_format(timestamp, 'yyyy-MM-dd') AS Date FROM search
)
,event_daily_stats AS
(
SELECT
u.Date
,u.user_id
,(SELECT COUNT(*) FROM search WHERE user_id=u.user_id AND date_format(timestamp, 'yyyy-MM-dd')=u.Date) AS searches
,(SELECT COUNT(*) FROM purchase WHERE user_id=u.user_id AND event='purchase_complete' AND date_format(timestamp, 'yyyy-MM-dd')=Date) AS purchases
,IFNULL((SELECT SUM(amount) FROM purchase WHERE user_id=u.user_id AND event='purchase_complete' AND date_format(timestamp, 'yyyy-MM-dd')=Date),0) AS total_purchased_amount
FROM users AS u
ORDER BY u.Date ASC
)
,DataDistinct AS
(
SELECT DISTINCT Date FROM event_daily_stats
)
SELECT
dd.date
,COUNT(eds.user_id) AS total_users
,SUM(eds.searches) AS total_searches
,SUM(eds.purchases) AS total_purchases
,SUM(eds.total_purchased_amount) AS total_purchased_amount
FROM DataDistinct AS dd
INNER JOIN event_daily_stats AS eds ON eds.Date=dd.Date
GROUP BY dd.date
ORDER BY dd.Date ASC
